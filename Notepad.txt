รายละเอียดโปรเจคต์: ร้านรองเท้า (รองเท้าแตะ/รองเท้าผ้าใบ)

1. โปรโมชัน (Promotions)
*สมาชิกใหม่: รับคูปองส่วนลด 50% 
*ซื้อจำนวนมาก: ซื้อตั้งแต่ 2 คู่ขึ้นไป ลดเพิ่มคู่ละ 10% 
*การจัดส่ง: บริการจัดส่งฟรีเมื่อซื้อสินค้าตามราคาที่กำหนด 
*สินค้าพิเศษ: โปรโมชันสำหรับรองเท้ารุ่นลิมิเต็ด (ตามช่วงเวลาที่กำหนด)
*โปรโมชันลดส่งท้ายปี สูงสุด 70%

2. โครงสร้างผู้ใช้งานและแผนก (User Roles & Departments)
*ลูกค้า: กลุ่มลูกค้าทั่วไป / ผู้ใช้งานหลักของระบบ
*เจ้าของแบรนด์ / เจ้าของร้าน / แอดมิน: ผู้ดูแลภาพรวมธุรกิจ 
*พนักงานสต็อกสินค้า: รับผิดชอบการเช็กและนับจำนวนสินค้า 
*พนักงานดูแลระบบ: จัดการดูแลความเรียบร้อยของระบบหลังบ้าน 
*พนักงานขาย: คิดโปรโมชัน
*พนักงานแพ็คสินค้า: จัดการด้านการขนส่งสินค้าให้ลูกค้า

3. ความต้องการด้านระบบ (System Requirements)

*ฝั่งลูกค้า (Customer Interface):
 **ระบบล็อคอิน: สมัครสมาชิกใหม่(Register)/Login เพื่อเข้าใช้งานระบบได้
 **ระบบแสดงสินค้า: แสดงตัวอย่างสินค้าและราคา (ทั้งราคาก่อนลดและราคาโปรโมชัน) 
 **ระบบชำระเงิน: รองรับการชำระผ่านธนาคารและบัตรเครดิต 
 **ระบบส่วนลด/คูปอง: สามารถกรอกคูปองหรือเลือกใช้ส่วนลดเพิ่มเติมได้ 
 **ระบบติดตาม: สามารถติดตามสถานะการชำระเงินหรือสถานะสินค้าได้ 
 **ระบบการเข้าดูสินค้าล่าสุด: สามารถดูได้ว่าปัจจุบันสั่งสินค้าอะไรไว้บ้าง
 **ระบบค้นหา: ค้นหาสินค้าและแยกหมวดหมู่สินค้าได้อย่างชัดเจน 

*ฝั่งร้านค้า/หลังบ้าน (Admin & Management):
 **ระบบสต็อก: ระบบจัดการสินค้าคงคลังแบบ Real-time 
 **ระบบบัญชี: ระบบแบ่งการทำงานของพนักงานแต่ละคน ให้สามารถจัดการได้เฉพาะหมวดหมู่หน้าที่ของตัวเองเท่านั้น 
 **ระบบพยากรณ์: วิเคราะห์แนวโน้มการขายสินค้า 
 **ระบบสรุปยอด: สรุปยอดขายแบบรายเดือนและรายปี 
 **ระบบสรุปสินค้ายอดฮิต: สรุปรายการสินค้ายอดนิยม

CREATE TABLE roles (
    id INT AUTO_INCREMENT PRIMARY KEY,
    role_name VARCHAR(50) NOT NULL UNIQUE
);

CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,
    role_id INT NOT NULL,
    fullname VARCHAR(100) NOT NULL,
    email VARCHAR(100) NOT NULL UNIQUE,
    password_hash VARCHAR(255) NOT NULL,
    phone VARCHAR(20),
    address TEXT,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (role_id) REFERENCES roles(id)
);

CREATE TABLE categories (
    id INT AUTO_INCREMENT PRIMARY KEY,
    category_name VARCHAR(100) NOT NULL
);

CREATE TABLE products (
    id INT AUTO_INCREMENT PRIMARY KEY,
    category_id INT NOT NULL,
    product_name VARCHAR(150) NOT NULL,
    description TEXT,
    price DECIMAL(10,2) NOT NULL,
    discount_percent DECIMAL(5,2) DEFAULT 0,
    is_limited BOOLEAN DEFAULT FALSE,
    stock_total INT DEFAULT 0,
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (category_id) REFERENCES categories(id)
);

CREATE TABLE product_variants (
    id INT AUTO_INCREMENT PRIMARY KEY,
    product_id INT NOT NULL,
    size VARCHAR(10),
    color VARCHAR(50),
    stock_quantity INT DEFAULT 0,
    FOREIGN KEY (product_id) REFERENCES products(id)
);

CREATE TABLE coupons (
    id INT AUTO_INCREMENT PRIMARY KEY,
    coupon_code VARCHAR(50) UNIQUE NOT NULL,
    discount_percent DECIMAL(5,2),
    min_purchase DECIMAL(10,2) DEFAULT 0,
    start_date DATETIME,
    end_date DATETIME
);

CREATE TABLE cart_items (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    product_variant_id INT NOT NULL,
    quantity INT NOT NULL,
    FOREIGN KEY (user_id) REFERENCES users(id),
    FOREIGN KEY (product_variant_id) REFERENCES product_variants(id)
);

CREATE TABLE orders (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    total_amount DECIMAL(10,2),
    discount_amount DECIMAL(10,2) DEFAULT 0,
    final_amount DECIMAL(10,2),
    coupon_id INT NULL,
    order_status VARCHAR(50) DEFAULT 'Pending',
    created_at DATETIME DEFAULT CURRENT_TIMESTAMP,
    FOREIGN KEY (user_id) REFERENCES users(id),
    FOREIGN KEY (coupon_id) REFERENCES coupons(id)
);

CREATE TABLE order_items (
    id INT AUTO_INCREMENT PRIMARY KEY,
    order_id INT NOT NULL,
    product_variant_id INT NOT NULL,
    price DECIMAL(10,2) NOT NULL,
    quantity INT NOT NULL,
    FOREIGN KEY (order_id) REFERENCES orders(id),
    FOREIGN KEY (product_variant_id) REFERENCES product_variants(id)
);

CREATE TABLE payments (
    id INT AUTO_INCREMENT PRIMARY KEY,
    order_id INT NOT NULL,
    payment_method VARCHAR(50),
    payment_status VARCHAR(50) DEFAULT 'Pending',
    paid_at DATETIME,
    FOREIGN KEY (order_id) REFERENCES orders(id)
);

CREATE TABLE shipments (
    id INT AUTO_INCREMENT PRIMARY KEY,
    order_id INT NOT NULL,
    tracking_number VARCHAR(100),
    shipping_status VARCHAR(50) DEFAULT 'Preparing',
    FOREIGN KEY (order_id) REFERENCES orders(id)
);
